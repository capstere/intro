Sub Duplicates()

Dim formatcount1 As Long
Dim formatcount2 As Long
Dim C As Range
Dim AC As Integer



Call ClearSheet

Application.ScreenUpdating = False

With ActiveSheet

    Range("C10").Columns.EntireColumn.Characters.Font.Color = vbBlack
    Range("C10").Columns.EntireColumn.Interior.ColorIndex = 0
    Range("C9:D9").Value = ""
    Range("C10").Select
    Range(Selection, Selection.End(xlDown)).Select
    Selection.FormatConditions.AddUniqueValues
    Selection.FormatConditions(Selection.FormatConditions.Count).SetFirstPriority
    Selection.FormatConditions(1).DupeUnique = xlDuplicate
    With Selection.FormatConditions(1).Font
        .Color = -16383844
        .TintAndShade = 0
    End With
    With Selection.FormatConditions(1).Interior
        .PatternColorIndex = xlAutomatic
        .Color = 13551615
        .TintAndShade = 0
    End With
    Selection.FormatConditions(1).StopIfTrue = True
    
    
    For Each C In Range("C10:C" & .Cells(.Rows.Count, "C").End(xlUp).Row)
        
       If C.DisplayFormat.Interior.Color = 13551615 Then
            
            Range("C9").Value = "Duplicate"
        
            With Range("C9")
            .HorizontalAlignment = xlCenter
            .VerticalAlignment = xlCenter
            .Font.Bold = True
            .Interior.Color = 13551615
            End With

            Range("C9").Characters.Font.Color = RGB(156, 0, 6)
            
       End If
    Next C
    
    '////////////////////////////////////////////////////////////////////////////////////////
    
    Range("D10").Select
    Range(Selection, Selection.End(xlDown)).Select
    Selection.FormatConditions.AddUniqueValues
    Selection.FormatConditions(Selection.FormatConditions.Count).SetFirstPriority
    Selection.FormatConditions(1).DupeUnique = xlDuplicate
    With Selection.FormatConditions(1).Font
        .Color = -16383844
        .TintAndShade = 0
    End With
    With Selection.FormatConditions(1).Interior
        .PatternColorIndex = xlAutomatic
        .Color = 13551615
        .TintAndShade = 0
    End With
    Selection.FormatConditions(1).StopIfTrue = True
    
    For Each C In Range("D10:D" & .Cells(.Rows.Count, "D").End(xlUp).Row)
        
       If C.DisplayFormat.Interior.Color = 13551615 Then

            Range("D9").Value = "Duplicate"
        
            With Range("D9")
            .HorizontalAlignment = xlCenter
            .VerticalAlignment = xlCenter
            .Font.Bold = True
            .Interior.Color = 13551615
            End With

            Range("D9").Characters.Font.Color = RGB(156, 0, 6)
            
       End If
    Next C
    
    
End With

Application.ScreenUpdating = True

Sub meanctcheck()
Application.ScreenUpdating = False

Dim lastRow As Long, HIVCol As Range
lastRow = Range("A65000").End(xlUp).Row

Dim assay As String, i As Long
assay = Range("A10").Value
i = 0

    'HIV-1 Ct > 39 Check
If assay = "Xpert_HIV-1 Qual" Then
    Set HIVCol = Range("AE10", "AE" & lastRow)
    For Each Cell In HIVCol
    If Cell.Value >= 39 Then
        Cell.EntireRow.Interior.Color = vbRed
        i = i + 1
    End If
    Next
End If
    
    'HIV-1 XC Ct Check
If assay = "Xpert HIV-1 Qual XC PQC" Then
    Set HIVCol = Range("AE10", "AE" & lastRow)
    For Each Cell In HIVCol
    If Cell.Value >= 38.8 Then
        Cell.EntireRow.Interior.Color = vbRed
        i = i + 1
    End If
    Next
End If

    Range("E5").Value = "CT Fail: " & i

    With Range("E5")
    .HorizontalAlignment = xlCenter
    .VerticalAlignment = xlCenter
    End With

    Range("E5").Characters.Font.Color = RGB(156, 0, 6)
    
End If


Application.ScreenUpdating = True

End Sub

            If DelamCheck = False Then

                    C.Interior.Color = RGB(198, 239, 206)
        
                End If
                
            ElseIf DelamCheck = True Then
            
                If DelamCID = "X" And Not SN Mod 2 > 0 Then
                    
                    C.Characters.Font.Color = RGB(156, 0, 6)
                    C.Interior.Color = RGB(255, 199, 206)
                    
                    i = i + 1
                    
                ElseIf DelamCID = "+" And Not SN Mod 2 = 0 Then
                    
                    C.Characters.Font.Color = RGB(156, 0, 6)
                    C.Interior.Color = RGB(255, 199, 206)
                    
                    i = i + 1
                    
                Else:
                
                    C.Characters.Font.Color = RGB(0, 97, 0)
                    C.Interior.Color = RGB(198, 239, 206)
                    
                End If
                
            End If
    Next C
    
End With

    With Range("C9")

    .VerticalAlignment = xlCenter
    End With

Sub IPT_CommaSplit_ColumnA_IfNeeded()
    Dim ws As Worksheet
    Dim lastRowA As Long
    Dim checkToRow As Long
    Dim r As Long
    Dim hasComma As Boolean
    Set ws = ActiveSheet
    lastRowA = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    If lastRowA < 1 Then Exit Sub
    'Om det redan finns data i kolumn B-Z, anta att CSV redan är uppdelad.
    If Application.WorksheetFunction.CountA(ws.Range("B1:Z" & lastRowA)) > 0 Then Exit Sub
    'Kolla bara de första 50 raderna efter kommatecken.
    checkToRow = lastRowA
    If checkToRow > 50 Then checkToRow = 50
    For r = 1 To checkToRow
        If InStr(1, CStr(ws.Cells(r, "A").Value), ",", vbBinaryCompare) > 0 Then
            hasComma = True
            Exit For
        End If
    Next r
    If hasComma = False Then Exit Sub
    'Motsvarar Data -> Text to Columns -> Delimited -> Comma -> Finish
    ws.Range("A1:A" & lastRowA).TextToColumns _
        Destination:=ws.Range("A1"), _
        DataType:=xlDelimited, _
        TextQualifier:=xlTextQualifierDoubleQuote, _
        ConsecutiveDelimiter:=False, _
        Tab:=False, _
        Semicolon:=False, _
        Comma:=True, _
        Space:=False, _
        Other:=False
End Sub
Sub Sorterman()
'
' Sorterman Macro
'
' Keyboard Shortcut: Ctrl+Shift+A
'
' Sorterar alla celler / Expanderar
IPT_CommaSplit_ColumnA_IfNeeded
Cells.Select
Cells.EntireColumn.AutoFit

Rows("10:10").Select
Range(Selection, Selection.End(xlDown)).Select
ActiveSheet.Sort.SortFields. _
Clear
ActiveSheet.Sort.SortFields. _
Add2 Key:=Range("C10", Range("C10").End(xlDown)), SortOn:=xlSortOnValues, Order:=xlAscending, _
DataOption:=xlSortNormal

.Header = xlGuess
.MatchCase = False
.Orientation = xlTopToBottom
.SortMethod = xlPinYin
.Apply
End With

End Sub
