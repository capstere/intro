            varLaggingDate1,
            If(

                DateAdd(varToday, -3, TimeUnit.Days),
                DateAdd(varToday, -1, TimeUnit.Days)

        Set(varLaggingDate2, If(varIsMonday, DateAdd(varToday, -2, TimeUnit.Days), Blank()));
        Set(varLaggingDate3, If(varIsMonday, DateAdd(varToday, -1, TimeUnit.Days), Blank()));

        // Load existing Delivery row for today
        Set(varDelivery_Today, LookUp(QDIP_Delivery_Daily, InputDate = varToday));
        // Load existing rows for laggingDate1
        Set(varEHS_Lag1, LookUp(QDIP_EHS_Daily, InputDate = varLaggingDate1));
        Set(varQ_Lag1, LookUp(QDIP_Quality_Daily, InputDate = varLaggingDate1));
        Set(varDelivery_Lag1, LookUp(QDIP_Delivery_Daily, InputDate = varLaggingDate1));
        // Samma för Lag2 och Lag3 när det är måndag

        Set(varEHS_Lag2, LookUp(QDIP_EHS_Daily, InputDate = varLaggingDate2));
        Set(varQ_Lag2, LookUp(QDIP_Quality_Daily, InputDate = varLaggingDate2));
